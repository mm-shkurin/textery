16. SQL Views для воронок
Registration Funnel
CREATE VIEW registration_funnel AS
SELECT
event_name,
COUNT(DISTINCT user_id)
FROM analytics_events
GROUP BY event_name;

Subscription Funnel
CREATE VIEW subscription_funnel AS
SELECT
event_name,
COUNT(DISTINCT user_id)
FROM analytics_events
WHERE event_name IN (
'PRICING_OPENED',
'CHECKOUT_STARTED',
'SUBSCRIPTION_ACTIVATED'
)
GROUP BY event_name;
